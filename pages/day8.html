Reference Solution (Check Your Work)

Try the challenge yourself first. Then compare against this reference that fills in every TODO — it pulls channel, RSSI, and rate from radiotap, tracks BSSIDs, and counts anomalies:

python pcap_analyzer_solution.py 
#!/usr/bin/env python3
from scapy.all import rdpcap, Dot11, Dot11Beacon, Dot11Deauth, \
    Dot11Disas, RadioTap
from collections import defaultdict, Counter
import sys

def rssi_bucket(dbm):
    if dbm > -50:  return "> -50 dBm (strong)"
    if dbm > -70:  return "-50 to -70 dBm (ok)"
    return "< -70 dBm (weak)"

def analyze_pcap(pcap_file):
    packets = rdpcap(pcap_file)
    types    = Counter()
    channels = Counter()
    rssi     = Counter()
    rates    = Counter()
    bssids   = defaultdict(int)
    anomalies = Counter()
    TYPE = {0: "Management", 1: "Control", 2: "Data"}

    for pkt in packets:
        if not pkt.haslayer(Dot11):
            continue
        types[TYPE.get(pkt.type, "Other")] += 1

        rt = pkt.getlayer(RadioTap)
        if rt is not None:
            if getattr(rt, "Channel", None):
                channels[rt.Channel] += 1
            if getattr(rt, "dBm_AntSignal", None) is not None:
                rssi[rssi_bucket(rt.dBm_AntSignal)] += 1
            if getattr(rt, "Rate", None):
                rates[f"{rt.Rate} Mbps"] += 1

        if pkt.haslayer(Dot11Beacon):
            bssids[pkt.addr3] += 1
        if pkt.haslayer(Dot11Deauth):
            anomalies["Deauthentications"] += 1
        if pkt.haslayer(Dot11Disas):
            anomalies["Disassociations"] += 1

    n = len(packets)
    print("=== PCAP Analysis Report ===")
    print(f"File: {pcap_file} ({n} frames)\n")
    print("Frame Summary:")
    for t, c in types.most_common():
        print(f"  {t}: {c} ({c/n*100:.1f}%)")
    print("\nSignal Strength:")
    for b, c in rssi.most_common():
        print(f"  {b}: {c}")
    print("\nTop BSSIDs (by beacons):")
    for bssid, c in sorted(bssids.items(), key=lambda x: -x[1])[:5]:
        print(f"  {bssid}: {c} beacons")
    print("\nAnomalies:")
    for a, c in anomalies.most_common():
        print(f"  {a}: {c}")

if __name__ == "__main__":
    if len(sys.argv) < 2:
        sys.exit("Usage: python pcap_analyzer_solution.py ")
    analyze_pcap(sys.argv[1])

🧠 Knowledge Check
Which capture layer provides channel, RSSI, and rate?

The radiotap header.

How do you classify a frame's category in code?

pkt.type (0 mgmt, 1 control, 2 data) plus the subtype.

Name two anomalies a Phase 1 analyzer should flag.

Deauth/disassoc bursts and FCS-bad (corrupted) frames.
