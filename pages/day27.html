The Goal: Explain the Efficiency Gap

A link may negotiate 866 Mbps yet deliver 400. The difference is real overhead: contention, inter-frame spacing, ACKs, retries, management traffic, and protection. Your capstone quantifies this by measuring both numbers from a single capture.

PHY rate ≠ throughput.
Rule of thumb: usable TCP throughput is roughly 45–60% of the negotiated PHY rate for a single client. Aggregation (A-MPDU) pushes it higher; retries and mixed legacy clients push it lower. Your tool makes this concrete for a real capture.

The Throughput Analyzer

python vht_throughput.py 
from scapy.all import rdpcap, RadioTap, Dot11
from collections import defaultdict

MCS = {0:(1,1/2),1:(2,1/2),2:(2,3/4),3:(4,1/2),4:(4,3/4),
       5:(6,2/3),6:(6,3/4),7:(6,5/6),8:(8,3/4),9:(8,5/6)}
N_SD = {20:52, 40:108, 80:234, 160:468}

def phy_rate(mcs, bw, nss, sgi=True):
    bpscs, r = MCS[mcs]
    t = 3.2e-6 if sgi else 3.6e-6
    return N_SD[bw] * bpscs * r * nss / t / 1e6   # Mbps

def analyze(pcap):
    pkts = rdpcap(pcap)
    t0, t1 = float(pkts[0].time), float(pkts[-1].time)
    duration = max(t1 - t0, 1e-6)

    data_bytes = 0
    rate_samples = []
    for p in pkts:
        if p.haslayer(Dot11) and p.type == 2:      # data frame
            data_bytes += len(p)
        rt = p.getlayer(RadioTap)
        if rt is not None and hasattr(rt, "MCS_index"):
            # bw/nss may need driver-specific extraction
            rate_samples.append(getattr(rt, "MCS_index"))

    actual = data_bytes * 8 / duration / 1e6       # Mbps
    print(f"Capture duration : {duration:.2f} s")
    print(f"Data bytes       : {data_bytes:,}")
    print(f"Actual throughput: {actual:.1f} Mbps")
    if rate_samples:
        common = max(set(rate_samples), key=rate_samples.count)
        theo = phy_rate(common, 80, 2)             # assume 80MHz/2SS
        print(f"Most common MCS  : {common}")
        print(f"Theoretical PHY  : {theo:.1f} Mbps")
        print(f"Efficiency       : {actual / theo * 100:.0f}%")

if __name__ == "__main__":
    import sys
    analyze(sys.argv[1])
