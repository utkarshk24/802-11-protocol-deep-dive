The Complete Connection Timeline

A successful WPA2/WPA3 connection is a precise sequence. Your analyzer reconstructs it and times each transition:

text Connection Lifecycle 
1. Probe Request / Response   discovery
2. Auth (seq 1)               STA -> AP
3. Auth (seq 2)               AP  -> STA   (status 0)
4. Assoc Request              STA -> AP
5. Assoc Response             AP  -> STA   (status 0, AID assigned)
6. EAPOL 1/4                  AP  -> STA   (ANonce)
7. EAPOL 2/4                  STA -> AP    (SNonce + MIC)
8. EAPOL 3/4                  AP  -> STA   (GTK + MIC)
9. EAPOL 4/4                  STA -> AP    (ACK)
   -> Data traffic flows (encrypted)

The Capstone Analyzer

python assoc_flow.py 
from scapy.all import rdpcap, Dot11, Dot11Auth, Dot11AssoReq, \
    Dot11AssoResp, Dot11ProbeReq, EAPOL

def analyze(pcap, client_mac):
    stages = []   # (label, time)
    for pkt in rdpcap(pcap):
        if not pkt.haslayer(Dot11):
            continue
        if client_mac not in (pkt.addr1, pkt.addr2):
            continue
        t = float(pkt.time)

        if pkt.haslayer(Dot11ProbeReq):
            stages.append(("Probe Request", t))
        elif pkt.haslayer(Dot11Auth):
            stages.append((f"Auth seq {pkt[Dot11Auth].seqnum}", t))
        elif pkt.haslayer(Dot11AssoReq):
            stages.append(("Assoc Request", t))
        elif pkt.haslayer(Dot11AssoResp):
            aid = pkt[Dot11AssoResp].AID & 0x3fff
            stages.append((f"Assoc Response (AID {aid})", t))
        elif pkt.haslayer(EAPOL):
            stages.append(("EAPOL handshake msg", t))

    if not stages:
        print("No frames for that client."); return

    t0 = stages[0][1]
    print(f"{'Stage':30s} {'t+ms':>8s}")
    print("-" * 40)
    for label, t in stages:
        print(f"{label:30s} {(t - t0) * 1000:8.1f}")

    total = (stages[-1][1] - t0) * 1000
    print("-" * 40)
    print(f"{'TOTAL connect time':30s} {total:8.1f} ms")

    eapol = sum(1 for l, _ in stages if l.startswith('EAPOL'))
    if eapol < 4:
        print(f"⚠️  Only {eapol}/4 EAPOL messages — incomplete handshake")

if __name__ == "__main__":
    import sys
    analyze(sys.argv[1], sys.argv[2])   # python assoc_flow.py cap.pcap aa:bb:..
