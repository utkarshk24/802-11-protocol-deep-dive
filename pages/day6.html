Scapy: Reading and Parsing 802.11

Installation

pip install scapy

Reading a PCAP File

python parse_frames.py 
from scapy.all import rdpcap, Dot11

# Read PCAP file
packets = rdpcap("capture.pcap")

# Iterate through frames
for pkt in packets:
    # Check if 802.11 frame
    if pkt.haslayer(Dot11):
        frame = pkt[Dot11]
        
        # Access Frame Control fields
        print(f"Frame Type: {frame.type}, Subtype: {frame.subtype}")
        print(f"From DS: {frame.fromds}, To DS: {frame.tods}")
        print(f"Protected: {frame.protected}")
        
        # Access MAC addresses
        print(f"RA: {frame.addr1} (Receiver)")
        print(f"TA: {frame.addr2} (Transmitter)")
        print(f"BSSID: {frame.addr3} (BSSID)")
        if frame.addr4:
            print(f"SA: {frame.addr4} (Source)")
        print()

Filtering Frames by Type

python parse_frames.py 
# Filter only management frames (type 0)
mgmt_frames = [pkt for pkt in packets if pkt.haslayer(Dot11) and pkt[Dot11].type == 0]

# Filter only beacons (subtype 8)
beacons = [pkt for pkt in mgmt_frames if pkt[Dot11].subtype == 8]

# Filter deauth frames (subtype 12)
deauths = [pkt for pkt in mgmt_frames if pkt[Dot11].subtype == 12]

# Count deauths by AP (BSSID)
from collections import defaultdict
deauth_by_ap = defaultdict(int)
for pkt in deauths:
    bssid = pkt[Dot11].addr3
    deauth_by_ap[bssid] += 1

# Alert if any AP sent >100 deauths
for bssid, count in deauth_by_ap.items():
    if count > 100:
        print(f"ALERT: AP {bssid} sent {count} deauths!")

Extracting Radiotap Information

If the PCAP was captured with Radiotap headers (most tools do), you can extract PHY info:

python parse_frames.py 
from scapy.layers.dot11 import Dot11Elt
from scapy.contrib.radiotap import Radiotap

for pkt in packets:
    if pkt.haslayer(Radiotap):
        radiotap = pkt[Radiotap]
        
        # Extract PHY layer info
        if hasattr(radiotap, 'dBm_AntSignal'):
            rssi = radiotap.dBm_AntSignal
            print(f"RSSI: {rssi} dBm")
        
        if hasattr(radiotap, 'Rate'):
            rate_mbps = radiotap.Rate / 2  # Scapy stores as 500 kbps units
            print(f"Data Rate: {rate_mbps} Mbps")

Real Example: Finding Failed Associations

Goal: Find all association failures and their reasons.

python find_assoc_failures.py 
from scapy.all import rdpcap, Dot11
from collections import defaultdict

packets = rdpcap("capture.pcap")

# Assoc response status codes
STATUS_CODES = {
    0: "Success",
    1: "Unspecified failure",
    10: "Cannot support all requested capabilities",
    11: "Reassociation no association",
    13: "Not authenticated",
    14: "Unknown authentication algorithm",
    15: "Unknown sequence number",
    16: "Challenge failure",
    17: "Timeout",
    25: "Invalid information element",
    27: "Invalid group cipher",
    28: "Invalid pairwise cipher",
}

failures = defaultdict(list)

for pkt in packets:
    if not pkt.haslayer(Dot11):
        continue
    
    frame = pkt[Dot11]
    
    # Look for assoc response (subtype 1)
    if frame.type == 0 and frame.subtype == 1:
        # Extract status code (2 bytes after frame control)
        # Scapy parses this as 'status'
        if hasattr(frame, 'status'):
            status_code = frame.status
            
            if status_code != 0:  # Not success
                bssid = frame.addr3
                reason = STATUS_CODES.get(status_code, f"Unknown ({status_code})")
                failures[reason].append({
                    'frame': pkt,
                    'bssid': bssid,
                    'status': status_code
                })

# Report failures
print("Association Failures:")
for reason, cases in failures.items():
    print(f"\n{reason} ({len(cases)} cases):")
    for case in cases[:5]:  # Show first 5
        print(f"  BSSID: {case['bssid']}")
