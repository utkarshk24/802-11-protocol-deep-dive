The Compliance Ruleset

text WPA3 Best-Practice Checks 
PASS/FAIL checks per BSS:
  [1] Uses SAE (WPA3) rather than PSK-only          (critical)
  [2] Management Frame Protection required (MFPR)   (critical)
  [3] TKIP disabled (CCMP/GCMP only)                (critical)
  [4] Not running WPA3/WPA2 transition mode         (warning)
  [5] No WEP / Open (unencrypted) network           (critical)
  [6] BSSID is on the approved allowlist            (warning)
  [7] Strong cipher (GCMP-256 for high-security)    (info)

The Validator

python wpa3_validator.py 
from scapy.all import rdpcap, Dot11, Dot11Beacon, Dot11Elt

# RSN AKM suite selectors (OUI 00-0F-AC)
AKM_PSK, AKM_SAE = b"\x00\x0f\xac\x02", b"\x00\x0f\xac\x08"
CIPHER_TKIP, CIPHER_CCMP = b"\x00\x0f\xac\x02", b"\x00\x0f\xac\x04"
CIPHER_GCMP256 = b"\x00\x0f\xac\x09"

def get_rsn(pkt):
    elt = pkt.getlayer(Dot11Elt)
    while elt:
        if elt.ID == 48:
            return bytes(elt.info)
        elt = elt.payload.getlayer(Dot11Elt)
    return None

def audit_bss(ssid, rsn, allowlist, bssid):
    checks, score = [], 0
    def add(ok, sev, msg):
        nonlocal score
        checks.append((("PASS" if ok else sev), msg))
        if ok: score += 1

    if rsn is None:
        checks.append(("CRITICAL", "Open/unencrypted network"))
        return checks, 0

    add(AKM_SAE in rsn,   "CRITICAL", "SAE (WPA3) enabled")
    add(AKM_PSK not in rsn or AKM_SAE in rsn,
                          "WARNING",  "Not PSK-only / transition mode")
    add(CIPHER_TKIP not in rsn, "CRITICAL", "TKIP disabled")
    add(CIPHER_CCMP in rsn or CIPHER_GCMP256 in rsn,
                          "CRITICAL", "Strong cipher (CCMP/GCMP)")
    # MFP required bit lives in RSN Capabilities (byte offset varies);
    # simplified: check presence of the capabilities field
    add(len(rsn) >= 20,   "WARNING",  "RSN Capabilities present (MFP)")
    add(bssid in allowlist.get(ssid, {bssid}),
                          "WARNING",  "BSSID on allowlist")
    return checks, score

def validate(pcap, allowlist=None):
    allowlist = allowlist or {}
    seen = {}
    for pkt in rdpcap(pcap):
        if pkt.haslayer(Dot11Beacon):
            ssid = pkt[Dot11Elt].info.decode(errors="ignore") or "<hidden>"
            seen[(ssid, pkt.addr3)] = get_rsn(pkt)

    print(f"{'='*60}\n WPA3 COMPLIANCE REPORT\n{'='*60}")
    for (ssid, bssid), rsn in seen.items():
        checks, score = audit_bss(ssid, rsn, allowlist, bssid)
        grade = "PASS" if all(c[0] == "PASS" for c in checks) else "REVIEW"
        print(f"\nSSID: {ssid}  [{bssid}]   -> {grade} ({score}/{len(checks)})")
        for status, msg in checks:
            icon = "✓" if status == "PASS" else "✗"
            print(f"   [{icon}] {status:9s} {msg}")

if __name__ == "__main__":
    import sys
    validate(sys.argv[1])
